# Fase M1.5 — Parser de Resposta OpenAI

## Objetivo

Extrair funções puras que saibam:
1. Ler uma resposta da API OpenRouter (formato OpenAI) e separar `content` textual de `tool_calls`.
2. Montar a mensagem `role: "tool"` de retorno no formato esperado.
3. Fazer parse defensivo dos `arguments` (que vêm como string JSON) sem lançar exceção.

Essas funções serão usadas pelo loop do agente em M3.

## Por que esta fase é separada

O formato de resposta do OpenAI/OpenRouter tem peculiaridades:
- `tool_calls` pode estar ausente, ser `null`, ou ser array vazio.
- `arguments` dentro de cada tool_call é uma **string JSON**, precisa de `JSON.parse`.
- `finish_reason` não é confiável para distinguir "tem tool_calls" vs "concluído" — alguns modelos mandam `stop` junto com `tool_calls`, ou `tool_calls` junto com `content` textual.
- Modelos diferentes via OpenRouter têm variações de comportamento.

Extraindo isso para funções puras, fica trivial testar todos os casos sem chamar a API.

## Detalhes de implementação

### `extractToolCalls`

```js
export function extractToolCalls(message) {
  if (!message || !Array.isArray(message.tool_calls) || message.tool_calls.length === 0) {
    return [];
  }
  return message.tool_calls.map((tc) => ({
    id: tc.id,
    name: tc.function?.name ?? "",
    arguments: tc.function?.arguments ?? "",
  }));
}
```

### `extractContent`

```js
export function extractContent(message) {
  if (!message) return null;
  const c = message.content;
  if (c === null || c === undefined || c === "") return null;
  return typeof c === "string" ? c : JSON.stringify(c);
}
```

> Nota: alguns modelos mandam `content` como array de blocos (`[{type:"text", text:"..."}]`). Para v1, normalizamos para string via `JSON.stringify` — pode ser refinado depois se necessário.

### `parseToolArgs`

```js
export function parseToolArgs(rawArgs) {
  if (!rawArgs) return { args: {}, error: null };
  try {
    const parsed = JSON.parse(rawArgs);
    if (typeof parsed !== "object" || parsed === null || Array.isArray(parsed)) {
      return { args: {}, error: `argumentos não são um objeto JSON: ${rawArgs}` };
    }
    return { args: parsed, error: null };
  } catch (e) {
    return { args: {}, error: `argumentos inválidos: ${e.message}` };
  }
}
```

### `buildToolResultMessage`

```js
export function buildToolResultMessage(toolCallId, content) {
  return {
    role: "tool",
    tool_call_id: toolCallId,
    content: String(content),
  };
}
```

## Testes unitários — `test/parseResponse.test.js`

```js
import { describe, it, expect } from "vitest";
import {
  extractToolCalls,
  extractContent,
  parseToolArgs,
  buildToolResultMessage,
} from "../src/parseResponse.js";

describe("extractToolCalls", () => {
  it("retorna array vazio quando message não tem tool_calls", () => {
    expect(extractToolCalls({ content: "oi" })).toEqual([]);
  });

  it("retorna array vazio quando tool_calls é null", () => {
    expect(extractToolCalls({ content: "oi", tool_calls: null })).toEqual([]);
  });

  it("retorna array vazio quando tool_calls é array vazio", () => {
    expect(extractToolCalls({ tool_calls: [] })).toEqual([]);
  });

  it("extrai 1 tool_call normalizando formato", () => {
    const msg = {
      tool_calls: [
        { id: "call_1", function: { name: "read_file", arguments: '{"path":"a.txt"}' } },
      ],
    };
    expect(extractToolCalls(msg)).toEqual([
      { id: "call_1", name: "read_file", arguments: '{"path":"a.txt"}' },
    ]);
  });

  it("extrai múltiplos tool_calls na ordem", () => {
    const msg = {
      tool_calls: [
        { id: "a", function: { name: "read_file", arguments: "{}" } },
        { id: "b", function: { name: "write_file", arguments: "{}" } },
        { id: "c", function: { name: "run_bash", arguments: "{}" } },
      ],
    };
    const out = extractToolCalls(msg);
    expect(out).toHaveLength(3);
    expect(out.map((t) => t.id)).toEqual(["a", "b", "c"]);
  });

  it("lida com function ausente", () => {
    const msg = { tool_calls: [{ id: "x" }] };
    expect(extractToolCalls(msg)).toEqual([{ id: "x", name: "", arguments: "" }]);
  });

  it("lida com message null/undefined", () => {
    expect(extractToolCalls(null)).toEqual([]);
    expect(extractToolCalls(undefined)).toEqual([]);
  });
});

describe("extractContent", () => {
  it("retorna string quando content é string", () => {
    expect(extractContent({ content: "olá" })).toBe("olá");
  });

  it("retorna null quando content é null", () => {
    expect(extractContent({ content: null })).toBeNull();
  });

  it("retorna null quando content é undefined", () => {
    expect(extractContent({})).toBeNull();
  });

  it("retorna null quando content é string vazia", () => {
    expect(extractContent({ content: "" })).toBeNull();
  });

  it("retorna null quando message é null", () => {
    expect(extractContent(null)).toBeNull();
  });
});

describe("parseToolArgs", () => {
  it("faz parse de JSON válido", () => {
    expect(parseToolArgs('{"path":"a.txt"}')).toEqual({
      args: { path: "a.txt" },
      error: null,
    });
  });

  it("retorna args vazio e error null para string vazia", () => {
    expect(parseToolArgs("")).toEqual({ args: {}, error: null });
  });

  it("não lança para JSON inválido", () => {
    const { args, error } = parseToolArgs("{invalid json");
    expect(args).toEqual({});
    expect(error).toMatch(/argumentos inválidos/);
  });

  it("rejeita array JSON", () => {
    const { args, error } = parseToolArgs("[1,2,3]");
    expect(args).toEqual({});
    expect(error).toMatch(/não é um objeto/);
  });

  it("rejeita string JSON pura", () => {
    const { args, error } = parseToolArgs('"hello"');
    expect(args).toEqual({});
    expect(error).toMatch(/não é um objeto/);
  });

  it("rejeita null JSON", () => {
    const { args, error } = parseToolArgs("null");
    expect(args).toEqual({});
    expect(error).toMatch(/não é um objeto/);
  });
});

describe("buildToolResultMessage", () => {
  it("monta mensagem no formato esperado", () => {
    expect(buildToolResultMessage("call_1", "resultado")).toEqual({
      role: "tool",
      tool_call_id: "call_1",
      content: "resultado",
    });
  });

  it("converte content não-string para string", () => {
    const out = buildToolResultMessage("c", { x: 1 });
    expect(out.content).toBe("[object Object]");
  });

  it("aceita content numérico", () => {
    const out = buildToolResultMessage("c", 42);
    expect(out.content).toBe("42");
  });
});
```
